树状数组可以用于维护前后缀最值。

维护前缀最大值：

```cpp
struct bitPre{
	int t[N+1];
	int lowbit(int x){
		return x&-x;
	}
	void set(int x,int k){
		while(x<=N+3){
			k=max(t[x],k);
			t[x]=k;
			x+=lowbit(x);
		}
	}
	int query(int x){
		int ans=0;
		while(x){
			ans=max(ans,t[x]);
			x-=lowbit(x);
		} 
		return ans;
	}
};
```

struct bitSuf{
